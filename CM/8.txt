y[t_] := a0 + a1*Cos[t] + b1*Sin[t] + a2*Cos[2 t] + b2*Sin[2 t];
solY = Solve[{y[0] == 0, y[2 Pi/3] == 1, y[Pi] == 1.5, y[4 Pi/3] == 4,
     y[5 Pi/3] == 2}, {a0, a1, a2, b1, b2}];
y[t_] = y[t] /. solY[[1]]
points = ListPlot[{{0, 0}, {1, 1}, {1.5, 1.5}, {2, 4}, {2.5, 2}}, 
   PlotStyle -> Red];
(*plotX=Plot[x[t],{t,0,2.5}];*)
plotY = Plot[y[t], {t, 0, 20}, PlotStyle -> Green, PlotRange -> All];
Show[plotY, points]
