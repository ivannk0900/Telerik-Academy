f[x_] := a*(1/(1 + x)) + b*(1/(2 + x)) + c*(1/(3 + x)) + 
  d*(1/(4 + x)) + e*(1/(5 + x))
sol = Solve[{f[0] == 0.1, f[2] == 0.009, f[4] == 0.0011, 
   f[6] == 0.00003, f[8] == 0.0000012}, {a, b, c, d, e}]

f[x_] = f[x] /. sol[[1]]

plot1 = Plot[f[x], {x, -1.5, 9}]
plot2 = ListPlot[{{0, 0.1}, {2, 0.009}, {4, 0.0011}, {6, 0.00003}, {8,
     0.0000012}}, PlotMarkers -> \[FilledCircle]]
Show[plot1, plot2, PlotRange -> All]
