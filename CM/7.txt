Lagrange[n_, x0_, h_, f_] := 
 (
  nodes = Table[Subscript[x, i] = x0 + i*h, {i, 0, n}];
  Do[
   num = 1;
   den = 1;
   Do[If[i == k, num *= 1; den *= 1, num *= (x - Subscript[x, k]); 
     den *= (Subscript[x, i] - Subscript[x, k])],
    {k, 0, n}];
   Subscript[l, i] = num/den, {i, 0, n}];
  sum = 0;
  Do[sum += Subscript[l, i]*f[Subscript[x, i]]
   , {i, 0, n}];
  sum
  )

f[x_] = x^4 + 2

2 + x^4
