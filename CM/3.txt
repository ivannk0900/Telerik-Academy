l2[x_] = ((x - 0) (x - Pi/4))/((Pi/2 - 0) (Pi/2 - Pi/4))
p[x_] = 1*l0[x] + Cos[Pi/4]*l1[x] + Cos[Pi/2]*l2[x]
Plot[Abs[(Cos[x] - p[x])/Cos[x]], {x, 0, Pi/2}]
